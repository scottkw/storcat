## If the keyword is not defined, "wails_tools.nsh" will populate them with the values from ProjectInfo.
## If they are defined here, "wails_tools.nsh" will not touch them. This allows to use this project.nsi manually

## Then you can call makensis on this file with specifying the path to your binary:

## For a installer with both architectures:
## > makensis -DARG_WAILS_AMD64_BINARY=..\..\bin\app-amd64.exe -DARG_WAILS_ARM64_BINARY=..\..\bin\app-arm64.exe

####
## !define INFO_PROJECTNAME    "MyProject" # Default "{{.Name}}"

## !define PRODUCT_EXECUTABLE  "Application.exe"      # Default "${INFO_PROJECTNAME}.exe"

!define MUI_UNICON "..\icon.ico"

!define MUI_ABORTWARNING # This will warn the user if they exit from the installer.

!insertmacro MUI_PAGE_WELCOME # Welcome to the installer page.
# !insertmacro MUI_PAGE_LICENSE "resources\eula.txt" # Adds a EULA page to the installer
!insertmacro MUI_PAGE_DIRECTORY # In which folder install page.

#!uninstfinalize 'signtool --file "%1"'
#!finalize 'signtool --file "%1"'

Name "${INFO_PRODUCTNAME}"
OutFile "..\..\bin\${INFO_PROJECTNAME}-${ARCH}-installer.exe" # Name of the installer's file.
InstallDir "$PROGRAMFILES64\${INFO_COMPANYNAME}\${INFO_PRODUCTNAME}" # Default installing folder ($PROGRAMFILES is Program Files folder).
ShowInstDetails show # This will always show the installation details.

# EnVar plugin for safe PATH manipulation (no string length truncation)
!addplugindir "${__FILEDIR__}\Plugins"

Function .onInit
   !insertmacro wails.checkArchitecture
FunctionEnd

Section
    !insertmacro wails.setShellContext

    !insertmacro wails.webview2runtime

    SetOutPath $INSTDIR

    !insertmacro wails.files

    CreateShortcut "$SMPROGRAMS\${INFO_PRODUCTNAME}.lnk" "$INSTDIR\${PRODUCT_EXECUTABLE}"
    CreateShortCut "$DESKTOP\${INFO_PRODUCTNAME}.lnk" "$INSTDIR\${PRODUCT_EXECUTABLE}"

    !insertmacro wails.associateFiles
    !insertmacro wails.associateCustomProtocols

    !insertmacro wails.writeUninstaller

    ; --- PATH Registration (PKG-02) ---
    ; Add install directory to system PATH using EnVar plugin
    ; EnVar uses Win32 RegQueryValueEx directly -- no NSIS string length truncation
    ; AddValue is idempotent: no-op if $INSTDIR is already in PATH
    EnVar::SetHKLM
    EnVar::AddValue "PATH" "$INSTDIR"
    Pop $0  ; 0=success, non-zero=error

    ; Broadcast WM_SETTINGCHANGE so running shells pick up new PATH
    SendMessage ${HWND_BROADCAST} ${WM_SETTINGCHANGE} 0 "STR:Environment" /TIMEOUT=5000
SectionEnd

Section "uninstall"
    !insertmacro wails.setShellContext

    ; --- PATH Cleanup ---
    ; Remove install directory from system PATH before deleting files
    EnVar::SetHKLM
    EnVar::DeleteValue "PATH" "$INSTDIR"
    Pop $0  ; 0=success

    ; Broadcast WM_SETTINGCHANGE so running shells see PATH removal
    SendMessage ${HWND_BROADCAST} ${WM_SETTINGCHANGE} 0 "STR:Environment" /TIMEOUT=5000

    RMDir /r "$AppData\${PRODUCT_EXECUTABLE}" # Remove the WebView2 DataPath

    RMDir /r $INSTDIR

    Delete "$SMPROGRAMS\${INFO_PRODUCTNAME}.lnk"
    Delete "$DESKTOP\${INFO_PRODUCTNAME}.lnk"

    !insertmacro wails.unassociateFiles
    !insertmacro wails.unassociateCustomProtocols

    !insertmacro wails.deleteUninstaller
SectionEnd

The `Commands` field is WinGet metadata that informs `winget` of CLI commands the package provides. For `nullsoft` installer type, PATH registration is handled by the NSIS installer itself (Task 1) -- the `Commands` field is informational/discoverability only, not functional. But it is part of WinGet best practices and helps `winget` surface the package when users search for CLI tools.

Note: The `distribute.yml` sed command `s/2\.1\.0/${VERSION_CLEAN}/g` will NOT touch the `Commands` field since it contains no version string -- this is correct and requires no workflow changes.

```go
// GetCatalogHtmlPath returns the HTML file path for a catalog
func (a *App) GetCatalogHtmlPath(catalogPath string) (string, error) {
	var htmlPath string
	if filepath.Ext(catalogPath) == ".json" {
		htmlPath = catalogPath[:len(catalogPath)-5] + ".html"
	} else {
		htmlPath = catalogPath + ".html"
	}
	if _, err := os.Stat(htmlPath); err != nil {
		if os.IsNotExist(err) {
			return "", fmt.Errorf("HTML file not found: %s", htmlPath)
		}
		return "", fmt.Errorf("cannot access HTML file: %w", err)
	}
	return htmlPath, nil
}
```

Key changes from current implementation:
- Add `os.Stat(htmlPath)` call after deriving the path
- Return specific "HTML file not found" error for `os.IsNotExist`
- Return wrapped error for other `os.Stat` failures (permissions, etc.)
- `os` and `fmt` are already imported in app.go; `filepath` is already imported

Do NOT change the method signature `(string, error)` -- the envelope is constructed in the TypeScript shim, not the Go layer.
